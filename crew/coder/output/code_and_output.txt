def calculate_series(terms):
    total = 0.0
    for i in range(terms):
        term = 1 / (2 * i + 1)  # 1, 1/3, 1/5, 1/7, ...
        if i % 2 == 1:  # Subtract for every second term
            term = -term
        total += term
    return total * 4

# Calculate for the first 10,000 terms
result = calculate_series(10000)
print(f"Result: {result}")
